LanguageDefinition

Overview

Struct Name: TextEditor::LanguageDefinition

Represents the rules and settings for a specific programming language or text format. It defines how keywords, identifiers, comments, numbers, and other elements are highlighted, as well as other language-specific behaviors like case sensitivity or comment styles.

Constructor Summary

Constructor Description
LanguageDefinition() Default constructor that initializes mPreprocChar to '#', mAutoIndentation to true, mTokenize to nullptr, and mCaseSensitive to true.

Custom LanguageDefinition


        #include "TextEditor.h"
        #include "imgui.h"

        TextEditor editor;

        TextEditor::LanguageDefinition createCustomLanguage() {
            TextEditor::LanguageDefinition lang;
            lang.mName = "MyLang";
            lang.mCaseSensitive = true;
            lang.mAutoIndentation = true;
            lang.mPreprocChar = '@';

            lang.mSingleLineComment = "//";
            lang.mCommentStart = "/*";
            lang.mCommentEnd = "*/";

            lang.mKeywords.insert("if");
            lang.mKeywords.insert("else");
            lang.mKeywords.insert("while");
            lang.mKeywords.insert("return");
            lang.mKeywords.insert("func");

            lang.mPreprocIdentifiers["include"] = TextEditor::Identifier();
            lang.mPreprocIdentifiers["define"] = TextEditor::Identifier();

            lang.mIdentifiers["int"] = TextEditor::Identifier();
            lang.mIdentifiers["float"] = TextEditor::Identifier();
            lang.mIdentifiers["string"] = TextEditor::Identifier();

            lang.mTokenRegexStrings.push_back({ "\\b[0-9]+\\b", TextEditor::PaletteIndex::Number });
            lang.mTokenRegexStrings.push_back({ "\".*?\"", TextEditor::PaletteIndex::String });
            lang.mTokenRegexStrings.push_back({ "//.*", TextEditor::PaletteIndex::Comment });
            lang.mTokenRegexStrings.push_back({ "/\\*.*?\\*/", TextEditor::PaletteIndex::MultiLineComment });

            return lang;
        }

        void setup() {
            TextEditor::LanguageDefinition customLang = createCustomLanguage();
            editor.SetLanguageDefinition(customLang);

            editor.SetText(
                "// Custom language example\n"
                "func main() {\n"
                " int x = 42;\n"
                " float y = 3.14;\n"
                " string s = \"Hello World\";\n"
                " if (x > 0) {\n"
                " return;\n"
                " }\n"
                "}\n"
            );

            editor.SetPalette(TextEditor::GetDarkPalette());
            }

        void render() {
            ImGui::Begin("Editor with Custom Language");
            editor.Render("TextEditor Example");
            ImGui::End();
        }
